;-----------------------------------------------------------------------------------------------------------------------------------------
;# 설명 : filedb 내장 함수로 파일에서 정보를 가져와 반복 테스트 진행 
;-----------------------------------------------------------------------------------------------------------------------------------------

"$신규세션={host=localhost:4444,browserName=chrome,platform=WINDOWS}" 세션생성 한다.


; 테스트 데이터 파일 ("이름,생년월일" 구조)
"$데이터파일=|$GUITAR_ScriptPath|\|탤런트정보.txt" 설정한다.

; 파일의 총 Line 수를 가져옴
"$테스트대상수=Ubound(FileReadToArray('|$데이터파일|'))" 계산

; 3회 반복
"$i=1" 설정 하고, "$i=$테스트대상수" 까지 반복한다. {
	
	; filedbread 로 1줄 데이터를 가져옴
	"$테스트데이터=_Filedbread('|$데이터파일|',|$i|)" 계산한다.

	; "," 로 구문하여 첫번째 인자를 가져옴
	"$검색어=StringSplit ('|$테스트데이터|', ',')[1]" 계산한다.

	; "," 로 구문하여 두번째 인자를 가져옴
	"$생년월일=StringSplit ('|$테스트데이터|', ',')[2]" 계산한다.

	 Func_탤런트정보확인_WEBDRIVER.txt 을 실행한다. 

}

세션종료한다.
