;데이터 결과 저장 파일명
"$테스트결과파일=|$GUITAR_리포트경로|\testresult.csv" 설정 한다.

; 5회 반복
"$파일핸들=FileOpen('|$테스트결과파일|',512 + 2)" 계산한다.

$i=1 설정 $i=5 반복 {

	0.1초 대기한다.

	$테스트시간=$GUITAR_현재날짜와시간  설정한다.
	$테스트ID=$i 설정한다.
	$테스트결과=성공 설정한다.

	$i=3 같으면 $테스트결과=실패 설정한다.

	; 저장 내용
	"$저장내용=|$테스트시간|,|$테스트ID|,|$테스트결과|" 설정한다. 

	; 파일에 Line 출력

	"$저장결과=FileWriteLine(|$파일핸들|,'|$저장내용|')" 계산한다.
}
"$파일닫기=FileClose(|$파일핸들|)" 계산한다.


; 최종 생성된 csv 파일을 테스트 종료후 자동으로 열게 함
"$실행결과=ShellExecute('|$테스트결과파일|')" 계산한다.
